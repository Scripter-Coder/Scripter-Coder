-- Function to find player by partial or full username
local function findPlayerByName(username)
    if username == "" then return nil end
    
    username = string.lower(username)
    
    local players = game:GetService("Players"):GetPlayers()
    
    for _, player in ipairs(players) do
        if string.lower(player.Name) == username then
            return player
        end
    end
    
    for _, player in ipairs(players) do
        if string.lower(player.Name):find(username, 1, true) then
            return player
        end
    end
    
    for _, player in ipairs(players) do
        if string.lower(player.DisplayName):find(username, 1, true) then
            return player
        end
    end
    
    return nil
end

-- Function to teleport to player
local function teleportToPlayer(player)
    if not player then
        Fluent:Notify({
            Title = "Error",
            Content = "Player not found!",
            Duration = 3
        })
        return false
    end
    
    local localPlayer = game.Players.LocalPlayer
    local localCharacter = localPlayer.Character
    local targetCharacter = player.Character
    
    if not localCharacter then
        Fluent:Notify({
            Title = "Error",
            Content = "Your character not loaded!",
            Duration = 3
        })
        return false
    end
    
    if not targetCharacter then
        Fluent:Notify({
            Title = "Error",
            Content = player.Name .. "'s character not loaded!",
            Duration = 3
        })
        return false
    end
    
    local targetHRP = targetCharacter:FindFirstChild("HumanoidRootPart") or targetCharacter:FindFirstChild("Torso")
    local localHRP = localCharacter:FindFirstChild("HumanoidRootPart") or localCharacter:FindFirstChild("Torso")
    
    if not targetHRP then
        Fluent:Notify({
            Title = "Error",
            Content = player.Name .. " has no HumanoidRootPart!",
            Duration = 3
        })
        return false
    end
    
    if not localHRP then
        Fluent:Notify({
            Title = "Error",
            Content = "You have no HumanoidRootPart!",
            Duration = 3
        })
        return false
    end
    
    local targetHumanoid = targetCharacter:FindFirstChildOfClass("Humanoid")
    if targetHumanoid and targetHumanoid.Health <= 0 then
        Fluent:Notify({
            Title = "Warning",
            Content = player.Name .. " is dead!",
            Duration = 3
        })
    end
    
    local success = pcall(function()
        local teleportPos = targetHRP.Position + Vector3.new(0, 3, 0)
        localHRP.CFrame = CFrame.new(teleportPos)
    end)
    
    if success then
        Fluent:Notify({
            Title = "Success",
            Content = "Teleported to " .. player.Name,
            Duration = 3
        })
        return true
    else
        Fluent:Notify({
            Title = "Error",
            Content = "Failed to teleport!",
            Duration = 3
        })
        return false
    end
end

local BerserkAntiRagdollToggle = Tabs.Berserk:AddToggle("BerserkAntiRagdoll", {
    Title = "Anti Ragdoll",
    Default = false,
    Callback = function(Value)
        _G.AntiRagdoll = Value
        while _G.AntiRagdoll do
            if game.Players.LocalPlayer.Character:FindFirstChild("HumanoidRootPart") and game.Players.LocalPlayer.Character:FindFirstChild("Torso") and game.Players.LocalPlayer.Character:FindFirstChild("Ragdolled") then
                if game.Players.LocalPlayer.Character:FindFirstChild("Ragdolled") and game.Players.LocalPlayer.Character:WaitForChild("Ragdolled").Value == true then
                    repeat task.wait()
                        if game.Players.LocalPlayer.Character:FindFirstChild("Torso") then
                            game.Players.LocalPlayer.Character.Torso.Anchored = true
                        end
                    until game.Players.LocalPlayer.Character:FindFirstChild("Ragdolled") and game.Players.LocalPlayer.Character:WaitForChild("Ragdolled").Value == false
                    if game.Players.LocalPlayer.Character:FindFirstChild("Torso") then
                        game.Players.LocalPlayer.Character.Torso.Anchored = false
                    end
                end
            end
            task.wait()
        end
    end
})

-- Debug Tab
local debugCode = nil

Tabs.Debug:AddButton({
    Title = "Get Code",
    Callback = function()
        local code = ""

        local success1, duckCount = pcall(function()
            local ducks = workspace["Debug Room"].DuckTable.DuckTable:GetChildren()
            local count = 0
            for _, obj in ipairs(ducks) do
                if obj:IsA("MeshPart") and obj.Name == "Duck" then
                    count += 1
                end
            end
            return count
        end)
        code = code .. (success1 and tostring(duckCount) or "?")

        local success2, adminNum = pcall(function()
            local text = workspace["Debug Room"].AdminGloves.GlovesCode.SurfaceGui.AdminNumber.Text
            return tonumber(text) or 0
        end)
        code = code .. (success2 and tostring(adminNum) or "?")

        local success3, mazeNum = pcall(function()
            local text = workspace["Debug Room"].Maze.MazePrize.SurfaceGui.MazeNumber.Text
            return tonumber(text) or 0
        end)
        code = code .. (success3 and tostring(mazeNum) or "?")

        local success4, responseNum = pcall(function()
            local possiblePaths = {
                workspace:FindFirstChild("Rig"),
                workspace:FindFirstChild("Debug Room"),
                workspace:FindFirstChild("Map")
            }
            
            for _, location in ipairs(possiblePaths) do
                if location then
                    local dialog = location:FindFirstChild("Dialog")
                    if not dialog then
                        dialog = location:FindFirstChildWhichIsA("Dialog")
                    end
                    
                    if dialog then
                        local dialogChoices = dialog:GetChildren()
                        for _, choice in ipairs(dialogChoices) do
                            if choice:IsA("DialogChoice") then
                                local response = choice.ResponseDialog
                                if response then
                                    local found = string.match(response, "%d+")
                                    if found then
                                        return tonumber(found)
                                    end
                                end
                            end
                        end
                    end
                end
            end
            return 0
        end)

        if not success4 or responseNum == 0 then
            local success5, altResponseNum = pcall(function()
                for _, dialog in ipairs(workspace:GetDescendants()) do
                    if dialog:IsA("Dialog") then
                        for _, choice in ipairs(dialog:GetChildren()) do
                            if choice:IsA("DialogChoice") then
                                local response = choice.ResponseDialog
                                if response then
                                    local found = string.match(response, "%d+")
                                    if found then
                                        return tonumber(found)
                                    end
                                end
                            end
                        end
                    end
                end
                return 0
            end)
            
            if success5 and altResponseNum ~= 0 then
                responseNum = altResponseNum
                success4 = true
            end
        end

        code = code .. (success4 and tostring(responseNum) or "?")

        debugCode = code

        Fluent:Notify({
            Title = "Code Checker",
            Content = "The Debug Glove Code is: " .. code,
            Duration = 5
        })
    end
})

Tabs.Debug:AddButton({
    Title = "Claim Debug Glove",
    Callback = function()
        if not debugCode or string.find(debugCode, "?") then
            Fluent:Notify({
                Title = "Error",
                Content = "Code not found or incomplete. Please run the first button again.",
                Duration = 4
            })
            return
        end

        local buttonsModel = workspace["Debug Room"].Keypad.Buttons

        for digit in debugCode:gmatch("%d") do
            local buttonPart = buttonsModel:FindFirstChild(digit)
            if buttonPart then
                local clickDetector = buttonPart:FindFirstChildOfClass("ClickDetector")
                if clickDetector then
                    fireclickdetector(clickDetector)
                    wait(0.2)
                end
            end
        end

        local enterPart = buttonsModel:FindFirstChild("Enter")
        if enterPart then
            local enterClick = enterPart:FindFirstChildOfClass("ClickDetector")
            if enterClick then
                fireclickdetector(enterClick)
            end
        end
        
        Fluent:Notify({
            Title = "Success",
            Content = "Code entered: " .. debugCode,
            Duration = 4
        })
    end
})

Tabs.Settings:AddSection("Slap Aura")

do
    local RunService = game:GetService("RunService")
    local ReplicatedStorage = game:GetService("ReplicatedStorage")
    local Players = game:GetService("Players")
    
    local slapAuraConnection
    local lastHitTime = 0
    local currentCooldown = 0.5
    local isSlapAuraEnabled = false
    local localPlayer = Players.LocalPlayer
    
    local gloveHits = {
        ["b"] = ReplicatedStorage.b,
        ["GeneralHit"] = ReplicatedStorage.GeneralHit,
    }
    
    local function getCurrentGlove()
        if localPlayer:FindFirstChild("leaderstats") then
            local gloveValue = localPlayer.leaderstats:FindFirstChild("Glove")
            if gloveValue then
                return gloveValue.Value
            end
        end
        return "Default"
    end
    
    local function getGloveRemote(gloveName)
        local remote = gloveHits[gloveName]
        
        if remote and typeof(remote) == "Instance" and remote:IsA("RemoteEvent") then
            return remote
        end
        
        if remote and type(remote) == "string" then
            local pathRemote = ReplicatedStorage:FindFirstChild(remote, true)
            if pathRemote and pathRemote:IsA("RemoteEvent") then
                return pathRemote
            end
        end
        
        local variations = {
            gloveName .. "Hit",
            "hit" .. gloveName,
            gloveName .. "H",
            "Ht" .. gloveName,
            gloveName:upper() .. "Hit",
            gloveName:lower() .. "hit"
        }
        
        for _, variation in ipairs(variations) do
            local foundRemote = ReplicatedStorage:FindFirstChild(variation)
            if foundRemote and foundRemote:IsA("RemoteEvent") then
                return foundRemote
            end
        end
        
        if ReplicatedStorage:FindFirstChild("GeneralHit") then
            return ReplicatedStorage.GeneralHit
        else
            return ReplicatedStorage.b
        end
    end
    
    local function autoDetectGloveRemote(gloveName)
        local commonRemotes = {
            "b", "GeneralHit"
        }
        
        for _, remoteName in ipairs(commonRemotes) do
            if remoteName:lower():find(gloveName:lower()) or gloveName:lower():find(remoteName:lower()) then
                local remote = ReplicatedStorage:FindFirstChild(remoteName)
                if remote and remote:IsA("RemoteEvent") then
                    return remote
                end
            end
        end
        
        for _, child in pairs(ReplicatedStorage:GetChildren()) do
            if child:IsA("RemoteEvent") then
                local nameLower = child.Name:lower()
                local gloveLower = gloveName:lower()
                
                if nameLower:find(gloveLower) or nameLower:find("hit") or nameLower:find("slap") then
                    return child
                end
            end
        end
        
        return ReplicatedStorage:FindFirstChild("b") or ReplicatedStorage:FindFirstChild("GeneralHit")
    end

    local SlapAuraSpeedSlider = Tabs.Settings:AddSlider("SlapAuraSpeed", {
        Title = "Slap Aura Speed",
        Description = "Set the cooldown between slaps",
        Default = 0.5,
        Min = 0.1,
        Max = 2,
        Rounding = 1,
        Callback = function(Value)
            currentCooldown = Value
        end
    })
    
    local SlapAuraToggle = Tabs.Settings:AddToggle("SlapAura", {
        Title = "Slap Aura",
        Default = false,
        Callback = function(enabled)
            isSlapAuraEnabled = enabled
            
            if enabled then
                if slapAuraConnection then
                    slapAuraConnection:Disconnect()
                    slapAuraConnection = nil
                end
                
                local currentGlove = getCurrentGlove()
                local remote = getGloveRemote(currentGlove)
                
                if not remote or not remote:IsA("RemoteEvent") then
                    remote = autoDetectGloveRemote(currentGlove)
                    
                    if not remote or not remote:IsA("RemoteEvent") then
                        Fluent:Notify({
                            Title = "Slap Aura Error",
                            Content = "No remote found for glove: " .. currentGlove,
                            Duration = 3
                        })
                        isSlapAuraEnabled = false
                        return
                    end
                end
                
                slapAuraConnection = RunService.Heartbeat:Connect(function()
                    local currentTime = tick()
                    
                    if currentTime - lastHitTime < currentCooldown then
                        return
                    end
                    
                    if not isSlapAuraEnabled or not localPlayer.Character then
                        return
                    end
                    
                    local localChar = localPlayer.Character
                    local localRoot = localChar:FindFirstChild("HumanoidRootPart")
                    if not localRoot then
                        return
                    end
                    
                    for _, player in ipairs(Players:GetPlayers()) do
                        if player ~= localPlayer and player.Character then
                            local char = player.Character
                            local humanoid = char:FindFirstChildOfClass("Humanoid")
                            local torso = char:FindFirstChild("Torso") or char:FindFirstChild("UpperTorso")
                            
                            if torso and humanoid and humanoid.Health > 0 then
                                local playerRoot = char:FindFirstChild("HumanoidRootPart")
                                if playerRoot then
                                    local distance = (playerRoot.Position - localRoot.Position).Magnitude
                                    if distance > 40 then
                                        continue
                                    end
                                end
                                
                                local success = pcall(function()
                                    remote:FireServer(torso)
                                end)
                                
                                if success then
                                    lastHitTime = currentTime
                                    break
                                end
                            end
                        end
                    end
                end)
                
                Fluent:Notify({
                    Title = "Slap Aura Enabled",
                    Content = "Glove: " .. currentGlove,
                    Duration = 3
                })
                
            else
                if slapAuraConnection then
                    slapAuraConnection:Disconnect()
                    slapAuraConnection = nil
                end
                
                Fluent:Notify({
                    Title = "Slap Aura Disabled",
                    Content = "",
                    Duration = 2
                })
            end
        end
    })
