        if not localHRP then
            Fluent:Notify({ Title = "Error", Content = "Your character not found!", Duration = 2 })
            return
        end
        local closestPlayer = nil
        local closestDistance = math.huge
        local closestPos = nil
        for _, p in ipairs(game:GetService("Players"):GetPlayers()) do
            if p ~= LocalPlayer then
                local character = p.Character
                local hrp = character and character:FindFirstChild("HumanoidRootPart")
                if hrp then
                    local distance = (hrp.Position - localHRP.Position).Magnitude
                    if distance < closestDistance then
                        closestDistance = distance
                        closestPlayer = p
                        closestPos = hrp.Position
                    end
                end
            end
        end
        if closestPlayer and closestPos then
            localChar.HumanoidRootPart.CFrame = CFrame.new(closestPos)
            Fluent:Notify({ Title = "Teleported", Content = "Teleported to " .. closestPlayer.Name .. " (" .. math.floor(closestDistance) .. " studs away)", Duration = 2 })
        else
            Fluent:Notify({ Title = "Error", Content = "No other players found!", Duration = 2 })
        end
    end

-- Death Sound Modifier
local deathSoundEnabled = false
local currentSoundId = ""
local monitorConnection = nil
local childAddedConnections = {}

local function cleanupConnections()
    if monitorConnection then monitorConnection:Disconnect(); monitorConnection = nil end
    for _, conn in pairs(childAddedConnections) do pcall(function() conn:Disconnect() end) end
    childAddedConnections = {}
end

local function modifyDeathSounds(character)
    if not deathSoundEnabled or not character then return end
    local torso = character:FindFirstChild("Torso") or character:FindFirstChild("UpperTorso")
    if not torso then return end
    for _, sound in pairs(torso:GetChildren()) do
        if sound:IsA("Sound") then
            local soundName = sound.Name:lower()
            if soundName:find("death") or soundName:find("die") then
                pcall(function() if sound.SoundId ~= currentSoundId then sound.SoundId = currentSoundId end end)
            end
        end
    end
end

local function setupSoundListener(character)
    if not character then return end
    local torso = character:FindFirstChild("Torso") or character:FindFirstChild("UpperTorso")
    if not torso then return end
    local connection = torso.ChildAdded:Connect(function(child)
        if child:IsA("Sound") and deathSoundEnabled then
            local soundName = child.Name:lower()
            if soundName:find("death") or soundName:find("die") then
                task.wait(0.1)
                pcall(function() child.SoundId = currentSoundId end)
            end
        end
    end)
    table.insert(childAddedConnections, connection)
end

local function startMonitoring()
    if monitorConnection then return end
    monitorConnection = game:GetService("RunService").Heartbeat:Connect(function()
        if deathSoundEnabled then
            local character = game.Players.LocalPlayer.Character
            if character then modifyDeathSounds(character) end
        end
    end)
    local character = game.Players.LocalPlayer.Character
    if character then setupSoundListener(character) end
end

            if Site.nextPageCursor and Site.nextPageCursor ~= "null" and Site.nextPageCursor ~= nil then foundAnything = Site.nextPageCursor end
            for i, v in pairs(Site.data) do
                local Possible = true
                local ID = tostring(v.id)
                if tonumber(v.maxPlayers) > tonumber(v.playing) then
                    for _, Existing in pairs(AllIDs) do
                        if ID == tostring(Existing) then Possible = false; break end
                    end
                    if Possible then
                        table.insert(AllIDs, ID)
                        wait()
                        pcall(function()
                            writefile("NotSameServers.json", game:GetService('HttpService'):JSONEncode(AllIDs))
                            wait()
                            game:GetService("TeleportService"):TeleportToPlaceInstance(PlaceID, ID, game.Players.LocalPlayer)
                        end)
                        wait(4); break
                    end
                end
            end
        end
        local function Teleport()
            while wait() do
                pcall(function() TPReturner(); if foundAnything ~= "" then TPReturner() end end)
            end
        end
        Teleport()
    end
})

                            else touchPart = teleportPart:FindFirstChildWhichIsA("BasePart") end
                            if touchPart then
                                hrp.CFrame = touchPart.CFrame + Vector3.new(0, 3, 0); task.wait(0.1)

                                hrp.CFrame = touchPart.CFrame + touchPart.CFrame.LookVector * 3; task.wait(0.1)
                                for i = 1, 3 do
                                    hrp.CFrame = touchPart.CFrame - touchPart.CFrame.LookVector * 2; task.wait(0.05)
