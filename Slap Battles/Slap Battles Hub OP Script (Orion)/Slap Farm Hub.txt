-- Anti AFK function
local function setupAntiAfk()
    if antiAfkConnection then
        antiAfkConnection:Disconnect()
        antiAfkConnection = nil

-- Enable noclip
local function enableNoclip()
    if noclipConnection then
        noclipConnection:Disconnect()
    end
    
    noclipConnection = RunService.Stepped:Connect(function()
        if not character then 
            return 
        end
        
        for _, part in pairs(character:GetChildren()) do
            if part:IsA("BasePart") then
                part.CanCollide = false
            end
        end
    end)
end

-- Disable noclip
local function disableNoclip()
    if noclipConnection then
        noclipConnection:Disconnect()
        noclipConnection = nil
    end
    
    if character then
        for _, part in pairs(character:GetChildren()) do
            if part:IsA("BasePart") then
                part.CanCollide = true
            end
        end
    end
end

-- Walk to position using Humanoid:MoveTo
local function walkToPosition(targetPosition, stopDistance)
    if not character or not humanoid or not humanoidRootPart then return false end
    
    if noclipEnabled then
        enableNoclip()
    end
    
    humanoid.WalkSpeed = walkSpeed
    humanoid:MoveTo(targetPosition)
    
    local startTime = tick()
    local maxTime = 15
    
    while tick() - startTime < maxTime and dualBotActive do
        if not character or not humanoid or not humanoidRootPart then return false end
        
        local distance = (humanoidRootPart.Position - targetPosition).Magnitude
        
        if distance <= stopDistance then
            humanoid:MoveTo(humanoidRootPart.Position)
            return true
        end
        
        humanoid:MoveTo(targetPosition)
        task.wait(0.1)
    end
    
    humanoid:MoveTo(humanoidRootPart.Position)
    return false
end

-- Update character references
local function setupCharacter()
    character = player.Character
    if character then
        humanoid = character:FindFirstChild("Humanoid")
        humanoidRootPart = character:FindFirstChild("HumanoidRootPart")
        if humanoid then
            humanoid.WalkSpeed = walkSpeed
        end
        if noclipEnabled then
            enableNoclip()
        end
    end
end

-- Check if player is in arena (has entered)
local function isPlayerInArena(plr)
    if not plr.Character then return false end
    return plr.Character:FindFirstChild("entered") ~= nil
end

-- Check if we are in arena
local function amIInArena()
    return isPlayerInArena(player)
end

-- Walk to Teleport1 (for entering arena)
local function walkToTeleport1()
    local lobby = workspace:FindFirstChild("Lobby")
    if not lobby then 
        print("⚠️ Lobby not found!")
        return false 
    end
    
    local teleport1 = lobby:FindFirstChild("Teleport1")
    if not teleport1 or not teleport1:IsA("BasePart") then 
        print("⚠️ Teleport1 not found!")
        return false 
    end
    
    local teleportPos = teleport1.Position + Vector3.new(0, 3, 0)
    
    print("🚶 Walking to Teleport1...")
    
    return walkToPosition(teleportPos, 5)
end

-- Check if Dual glove is equipped
local function isDualEquipped()
    local leaderstats = player:FindFirstChild("leaderstats")
    if leaderstats then
        local glove = leaderstats:FindFirstChild("Glove")
        if glove then
            return glove.Value == "Dual"
        end
    end
    return false
end

-- Function to start auto using Dual tool (auto slaps when close)
local function startAutoUseDual()
    if isUsingDual then return end
    
    isUsingDual = true
    
    dualConnection = RunService.Heartbeat:Connect(function()
        if not dualBotActive then return end
        
        local character = player.Character
        if not character then return end
        
        local humanoidRootPart = character:FindFirstChild("HumanoidRootPart")
        if not humanoidRootPart then return end
        
        if not amIInArena() then return end
        
        local closestPlayer = nil
        local closestDistance = 5
        
        for _, plr in pairs(Players:GetPlayers()) do
            if plr ~= player and plr.Character then
                local hrp = plr.Character:FindFirstChild("HumanoidRootPart")
                if hrp and isPlayerInArena(plr) then
                    local shouldSkip = false
                    
                    if ignoreRocks and plr.Character:FindFirstChild("rock") then
                        shouldSkip = true
                    end
                    
                    if not shouldSkip then
                        local distance = (humanoidRootPart.Position - hrp.Position).Magnitude
                        if distance < closestDistance then
                            closestDistance = distance
                            closestPlayer = plr
                        end
                    end
                end
            end
        end
        
        if closestPlayer and closestDistance <= 5 then
            local dualTool = character:FindFirstChild("Dual") or player.Backpack:FindFirstChild("Dual")
            
            if dualTool then
                if dualTool.Parent == player.Backpack then
                    dualTool.Parent = character
                end
                
                if dualTool:IsA("Tool") then
                    pcall(function()
                        dualTool:Activate()
                    end)
                end
            end
        end
    end)
end

-- Function to stop auto using Dual tool
local function stopAutoUseDual()
    isUsingDual = false
    if dualConnection then
        dualConnection:Disconnect()
        dualConnection = nil
    end
end

-- Function to start auto jumping (random 1-10 seconds)
local function startAutoJumping()
    if jumpConnection then
        jumpConnection:Disconnect()
        jumpConnection = nil
    end
    
    if not enableJumping then return end
    
    nextJumpTime = tick() + math.random(1, 10)
    
    jumpConnection = RunService.Heartbeat:Connect(function()
        if not dualBotActive or not enableJumping then return end
        if not character or not humanoid then return end
        
        if amIInArena() then
            if tick() >= nextJumpTime then
                pcall(function()
                    humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
                end)
                nextJumpTime = tick() + math.random(1, 10)
                print("🦘 Jumped! Next jump in " .. math.floor(nextJumpTime - tick()) .. " seconds")
            end
        end
    end)
end

-- Function to stop auto jumping
local function stopAutoJumping()
    if jumpConnection then
        jumpConnection:Disconnect()
        jumpConnection = nil
    end
    nextJumpTime = 0
end

-- Barzil portal control
local function controlBarzilPortal()
    local brazilFolder = workspace:FindFirstChild("Lobby") and workspace.Lobby:FindFirstChild("brazil")
    if brazilFolder then
        for _, v in pairs(brazilFolder:GetDescendants()) do
            if v:IsA("BasePart") then
                v.CanTouch = not ignoreBarzilPortal
            end
        end
    end
end

-- Get CLOSEST player who has entered (excluding self) for walking
local function getClosestPlayerForWalking()
    if not character or not humanoidRootPart then return nil, math.huge end
    
    local closestPlayer = nil
    local closestDistance = math.huge
    
    for _, plr in pairs(Players:GetPlayers()) do
        if plr ~= player and plr.Character then
            local hrp = plr.Character:FindFirstChild("HumanoidRootPart")
            if hrp and isPlayerInArena(plr) then
                local shouldSkip = false
                
                if ignoreRocks and plr.Character:FindFirstChild("rock") then
                    shouldSkip = true
                end
                
                if not shouldSkip then
                    local distance = (humanoidRootPart.Position - hrp.Position).Magnitude
                    if distance < closestDistance then
                        closestDistance = distance
                        closestPlayer = plr
                    end
                end
            end
        end
    end
    
    return closestPlayer, closestDistance
end

-- Auto collect slapples - BRING METHOD
local function collectSlapples()

    local island5 = workspace:FindFirstChild("Arena") and workspace.Arena:FindFirstChild("island5")
    if not island5 then return end
    
    local slapplesFolder = island5:FindFirstChild("Slapples")
    if not slapplesFolder then return end
    
    for _, slappleModel in pairs(slapplesFolder:GetChildren()) do
        if slappleModel:IsA("Model") and (slappleModel.Name == "Slapple" or slappleModel.Name == "GoldenSlapple") then
            pcall(function()

                if glove and glove:IsA("BasePart") then
                    glove.CFrame = humanoidRootPart.CFrame
                end
            end)
        end
    end
end

-- Main Dual bot function
local function startDualBot()
    print("🔄 Dual Bot Started!")
    
    OrionLib:MakeNotification({
        Name = "Dual Bot Started",
        Content = "Walking mode enabled!",
        Image = "rbxassetid://4483345998",
        Time = 3
    })
    
    hasTeleportedToArena = false
    
    startAutoUseDual()
    
    if enableJumping then
        startAutoJumping()
    end
    
    while dualBotActive do
        if character and humanoid and humanoidRootPart then
            if not isDualEquipped() then
                OrionLib:MakeNotification({
                    Name = "Error",
                    Content = "Equip Dual glove!",
                    Image = "rbxassetid://7743878857",
                    Time = 3
                })
                dualBotActive = false
                break
            end
            
            if not amIInArena() then
                hasTeleportedToArena = false
                print("🚶 Not in arena, walking to Teleport1...")
                walkToTeleport1()
                task.wait(1)
                task.wait(0.5)
            else
                if not hasTeleportedToArena then
                    hasTeleportedToArena = true
                    print("✅ Entered arena!")
                    OrionLib:MakeNotification({
                        Name = "Entered",
                        Content = "Entered arena!",
                        Image = "rbxassetid://4483345998",
                        Time = 2
                    })
                end
                
                if ignoreBarzilPortal then
                    controlBarzilPortal()
                end
                
                if autoCollectSlapples then
                    task.spawn(function()
                        collectSlapples()
                    end)
                end
                
                local target, distanceToTarget = getClosestPlayerForWalking()
                
                if target and target.Character then
                    local targetHRP = target.Character:FindFirstChild("HumanoidRootPart")
                    if targetHRP then
                        local targetPos = targetHRP.Position
                        local currentDistance = (humanoidRootPart.Position - targetPos).Magnitude
                        
                        if currentDistance > 8 then
                            local walkPos = Vector3.new(targetPos.X, targetPos.Y + 3, targetPos.Z)
                            print("🚶 Walking to: " .. target.Name .. " (" .. math.floor(currentDistance) .. " studs away)")
                            walkToPosition(walkPos, 8)
                        else
                            humanoid:MoveTo(humanoidRootPart.Position)
                            
                            local lookAt = CFrame.new(humanoidRootPart.Position, targetPos)
                            humanoidRootPart.CFrame = lookAt
                        end
                    end
                else
                    if amIInArena() then
                        local safePos = safePositions["island5"] or safePositions["main island"]
                        if safePos then
                            print("🚶 No players found, walking to safe island...")
                            walkToPosition(safePos, 10)
                        end
                    end
                end
            end
        else
            setupCharacter()
        end
        
        task.wait(detectionSpeed)
    end
    
    stopAutoUseDual()
    stopAutoJumping()
    disableNoclip()
    if humanoid then
        humanoid.WalkSpeed = 16
    end
    
    print("🛑 Dual Bot Stopped!")
    
    OrionLib:MakeNotification({
        Name = "Dual Bot Stopped",
        Content = "Bot is now inactive",
        Image = "rbxassetid://4483345998",
        Time = 3
    })
end

-- Function to check if Brick Master badge is owned using PlayerData path
local function checkBrickMasterBadge()
    local playerData = replicatedStorage:FindFirstChild("PlayerData")
    if not playerData then
        ownBrickMasterBadge = false
        return false
    end
    
    local playerFolder = playerData:FindFirstChild(player.Name)
    if not playerFolder then
        ownBrickMasterBadge = false
        return false
    end
    
    local unlockedBadges = playerFolder:FindFirstChild("UnlockedBadges")
    if not unlockedBadges then
        ownBrickMasterBadge = false
        return false
    end
    
    if unlockedBadges:IsA("StringValue") then
        local badgeIdsString = unlockedBadges.Value
        
        if badgeIdsString and badgeIdsString ~= "" then
            local trimmed = badgeIdsString:sub(2, -2)
            
            for id in string.gmatch(trimmed, "[^,]+") do
                local cleanId = id:match("^%s*(.-)%s*$")
                if cleanId == BRICK_MASTER_BADGE_ID then
                    ownBrickMasterBadge = true
                    return true
                end
            end
        end
    end
    
    ownBrickMasterBadge = false
    return false
end

-- Function to format time EXACTLY LIKE MEGAROCK (Hours:Minutes:Seconds:Milliseconds)
local function formatTime(seconds, milliseconds)
    local hours = math.floor(seconds / 3600)
    local minutes = math.floor((seconds % 3600) / 60)
    local secs = math.floor(seconds % 60)
    local ms = math.floor(milliseconds or 0)
    return string.format("%02d:%02d:%02d:%03d", hours, minutes, secs, ms)
end

                    _G.AutoSlapReverse = false
                    return
                end
                
                -- Function to get all players with "Reversed" BoolValue
                local function getReversedPlayers()
                    local reversedPlayers = {}
                    
                    for _, player in ipairs(game.Players:GetPlayers()) do
                        if player ~= LocalPlayer then
                            -- Check workspace for player's model with Reversed BoolValue
                            local playerModel = workspace:FindFirstChild(player.Name)
                            if playerModel then
                                local reversedBool = playerModel:FindFirstChild("Reversed")
                                if reversedBool and reversedBool:IsA("BoolValue") and reversedBool.Value == true then
                                    table.insert(reversedPlayers, player)
                                end
                            end
                        end
                    end
                    
                    return reversedPlayers
                end
                
                -- Teleport to a random reversed player
                local function teleportToRandomReversedPlayer()
                    local reversedPlayers = getReversedPlayers()
                    
                    if #reversedPlayers > 0 then
                        local targetPlayer = reversedPlayers[math.random(1, #reversedPlayers)]
                        
                        if targetPlayer and targetPlayer.Character then
                            local targetHRP = targetPlayer.Character:FindFirstChild("HumanoidRootPart")
                            local localHRP = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
                            
                            if targetHRP and localHRP then
                                localHRP.CFrame = targetHRP.CFrame
                                return true
                            end
                        end
                    end
                    return false
                end
                
                -- Fire GeneralHit on a specific player
                local function fireGeneralHitOnPlayer(player)
                    if player and player.Character then
                        local rightArm = player.Character:FindFirstChild("Right Arm") or player.Character:FindFirstChild("RightHand")
                        if rightArm then
                            pcall(function()
                                ReplicatedStorage:WaitForChild("b"):FireServer(rightArm)
                            end)
                        end
                    end
                end
                
                -- Fire GeneralHit on all reversed players
                local function fireGeneralHitOnAllReversedPlayers()
                    local reversedPlayers = getReversedPlayers()
                    
                    for _, player in ipairs(reversedPlayers) do
                        fireGeneralHitOnPlayer(player)
                    end
                end
                
                -- Main loop
                while _G.AutoSlapReverse do
                    -- Get reversed players
                    local reversedPlayers = getReversedPlayers()
                    
                    if #reversedPlayers > 0 then
                        -- Teleport to random reversed player
                        teleportToRandomReversedPlayer()
                        task.wait(0.05) -- Small delay after teleport
                        
                        -- Hit all reversed players
                        fireGeneralHitOnAllReversedPlayers()
                    end

                end

                            local targetHRP = targetPlayer.Character:FindFirstChild("HumanoidRootPart")

                    end

                        teleportToRandomReversedPlayer()

    local names = getTycoonModelNames()

        end
    end
    return nil, nil

local function getCurrentPoints()
    local tycoon = findTycoon()

            if part then

                if surfaceGui then
                    local textLabel = surfaceGui:FindFirstChild("TextLabel")
                    if textLabel and textLabel:IsA("TextLabel") then
                        local text = textLabel.Text
                        local number = tonumber(text:match("%d+")) or 0
                        return number
                    end
                end
            end
        end
    end
    return 0
end

-- Function to check if SLAP BATTLES SWEEP badge is owned using PlayerData path
local function checkSlapBattlesSweepBadge()
    local playerData = replicatedStorage:FindFirstChild("PlayerData")
    if not playerData then
        ownSlapBattlesSweepBadge = false
        return false
    end
    
    local playerFolder = playerData:FindFirstChild(player.Name)
    if not playerFolder then
        ownSlapBattlesSweepBadge = false
        return false
    end
    
    local unlockedBadges = playerFolder:FindFirstChild("UnlockedBadges")
    if not unlockedBadges then
        ownSlapBattlesSweepBadge = false
        return false
    end
    
    if unlockedBadges:IsA("StringValue") then
        local badgeIdsString = unlockedBadges.Value
        
        if badgeIdsString and badgeIdsString ~= "" then
            local trimmed = badgeIdsString:sub(2, -2)
            
            for id in string.gmatch(trimmed, "[^,]+") do
                local cleanId = id:match("^%s*(.-)%s*$")
                if cleanId == SLAP_BATTLES_SWEEP_BADGE_ID then
                    ownSlapBattlesSweepBadge = true
                    return true
                end
            end
        end
    end
    
    ownSlapBattlesSweepBadge = false
    return false
end

local function RefreshPlayersList()
    PlayersList = {}
    for _, player in pairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            table.insert(PlayersList, player.Name)
        end
    end
    return PlayersList

local function GetGeneralHit()
    return ReplicatedStorage:FindFirstChild("GeneralHit")
end

local function SlapTarget(targetPart)
    local generalHit = GetGeneralHit()
    if generalHit and targetPart then
        generalHit:FireServer(targetPart)
        return true
    end
    return false
end

local function GetPlayerParts(player)
    if not player or not player.Character then return nil end
    
    local parts = {
        player.Character:FindFirstChild("Right Arm"),
        player.Character:FindFirstChild("Left Arm"),
        player.Character:FindFirstChild("Head"),
        player.Character:FindFirstChild("Torso"),
        player.Character:FindFirstChild("HumanoidRootPart")
    }
    
    -- Filter out nil parts
    local validParts = {}
    for _, part in pairs(parts) do
        if part then
            table.insert(validParts, part)
        end
    end
    
    -- Also search for any child named DOLLPART
    if player.Character then
        for _, child in pairs(player.Character:GetChildren()) do
            if child.Name == "DOLLPART" and child:IsA("BasePart") then
                table.insert(validParts, child)
            end
        end
    end
    
    return validParts
end

-- Store owned badges
local ownedBadges = {}

-- Fetch owned badges from ReplicatedStorage
local function fetchOwnedBadges()
    ownedBadges = {}
    
    local playerData = ReplicatedStorage:FindFirstChild("PlayerData")
    if not playerData then
        return false
    end
    
    local playerFolder = playerData:FindFirstChild(username)
    if not playerFolder then
        return false
    end
    
    local unlockedBadges = playerFolder:FindFirstChild("UnlockedBadges")
    if not unlockedBadges then
        return false
    end
    
    if unlockedBadges:IsA("StringValue") then
        local badgeIdsString = unlockedBadges.Value
        
        if badgeIdsString and badgeIdsString ~= "" then
            local trimmed = badgeIdsString:sub(2, -2)
            
            for id in string.gmatch(trimmed, "[^,]+") do
                local cleanId = id:match("^%s*(.-)%s*$")
                ownedBadges[cleanId] = true
            end
        end
        return true
    end
    return false
end

-- Helper function to get Null Shards
function WebhookSettings:GetNullShards()
    local playerData = ReplicatedStorage:FindFirstChild("PlayerData")
    if playerData then
        local userData = playerData:FindFirstChild(username)
        if userData then
            local nullShards = userData:FindFirstChild("NullShards")
            if nullShards then
                return nullShards.Value
            end
        end
    end
    return 0
end

-- Get ping prefix based on settings
local function getPingPrefix()
    if WebhookSettings.PingType == "No Pings" then
        return ""
    elseif WebhookSettings.PingType == "@User Ping" and WebhookSettings.UserID ~= "" then
        return "<@" .. WebhookSettings.UserID .. "> "
    elseif WebhookSettings.PingType == "@Role Ping" and WebhookSettings.RoleID ~= "" then
        return "<@&" .. WebhookSettings.RoleID .. "> "
    elseif WebhookSettings.PingType == "@Everyone Ping" then
        return "@everyone "
    elseif WebhookSettings.PingType == "@Here Ping" then
        return "@here "
    end
    return ""
end

-- Webhook send function
local function sendWebhookMessage(content)
    if not WebhookSettings.Enabled or WebhookSettings.WebhookURL == "" then return end
    
    local ping = getPingPrefix()
    local fullContent = ping .. content
    
    local payload = {
        content = fullContent or "No message provided"
    }
    
    local jsonData = HttpService:JSONEncode(payload)
    
    local success, response = pcall(function()
        if syn and syn.request then
            return syn.request({
                Url = WebhookSettings.WebhookURL,
                Method = "POST",
                Headers = {
                    ["Content-Type"] = "application/json"
                },
                Body = jsonData
            })
        elseif http_request then
            return http_request({
                Url = WebhookSettings.WebhookURL,
                Method = "POST",
                Headers = {
                    ["Content-Type"] = "application/json"
                },
                Body = jsonData
            })
        elseif request then
            return request({
                Url = WebhookSettings.WebhookURL,
                Method = "POST",
                Headers = {
                    ["Content-Type"] = "application/json"
                },
                Body = jsonData
            })
        else
            warn("No request method available for webhook!")
            return nil, "No request method available"
        end
    end)
    
    if not success then
        warn("Failed to send webhook: " .. tostring(response))
    end
end

-- Get badge name from ID
local function getBadgeName(badgeId)
    for _, badge in pairs(allBadges) do
        if badge.id == badgeId then
            return badge.name
        end
    end
    return "Unknown Badge"
end

-- Object detection function
local function checkObjectSpawns()
    if not WebhookSettings.Enabled then return end
    
    -- Regular objects (not Slapples)
    local objectsToCheck = {
        ["Jet Orb"] = workspace:FindFirstChild("JetOrb"),
        ["Phase Orb"] = workspace:FindFirstChild("PhaseOrb"),
        ["Glitch Orb"] = workspace:FindFirstChild("GlitchOrb"),
        ["Siphon Orb"] = workspace:FindFirstChild("SiphonOrb"),
        ["MATERIALIZE Orb"] = workspace:FindFirstChild("MATERIALIZEOrb"),
        ["Toolbox"] = workspace:FindFirstChild("Toolbox"),
    }
    
    for name, object in pairs(objectsToCheck) do
        if table.find(WebhookSettings.LoggedObjects, name) then
            if object and not WebhookSettings.ObjectLogged[name] then
                WebhookSettings.ObjectLogged[name] = true
                
                local placeName = "Unknown"
                pcall(function()
                    local info = MarketplaceService:GetProductInfo(game.PlaceId)
                    if info and info.Name then
                        placeName = info.Name
                    end
                end)
                
                local message = string.format(
                    "**From Webhook Log:**\n\n**Detected | Spawned:** %s\n\n**Name:** %s\n**Real Name:** %s\n**Place Name:** %s\n**Place ID:** %s\n**JobID:** %s",
                    name, name, object.Name, placeName, game.PlaceId, game.JobId
                )
                
                sendWebhookMessage(message)
            elseif not object then
                WebhookSettings.ObjectLogged[name] = false
            end
        end
    end
    
    -- SLAPPLE CHECK (workspace.Arena.island5.Slapples)
    if table.find(WebhookSettings.LoggedObjects, "Slapple") then
        local slapplesFolder = workspace:FindFirstChild("Arena")
        if slapplesFolder then
            local island5 = slapplesFolder:FindFirstChild("island5")
            if island5 then
                local slapples = island5:FindFirstChild("Slapples")
                if slapples then
                    for _, model in ipairs(slapples:GetChildren()) do
                        if model:IsA("Model") then
                            local gloveMesh = model:FindFirstChild("Glove")
                            if gloveMesh and gloveMesh:IsA("MeshPart") then
                                local transparency = gloveMesh.Transparency
                                local modelName = model.Name
                                
                                if not WebhookSettings.SlappleLogged[modelName] then
                                    if transparency == 0 then
                                        WebhookSettings.SlappleLogged[modelName] = true
                                        
                                        local placeName = "Unknown"
                                        pcall(function()
                                            local info = MarketplaceService:GetProductInfo(game.PlaceId)
                                            if info and info.Name then
                                                placeName = info.Name
                                            end
                                        end)
                                        
                                        local message = string.format(
                                            "**From Webhook Log:**\n\n**Detected | Spawned:** Slapple\n\n**Name:** Slapple\n**Real Name:** %s\n**Place Name:** %s\n**Place ID:** %s\n**JobID:** %s",
                                            modelName, placeName, game.PlaceId, game.JobId
                                        )
                                        
                                        sendWebhookMessage(message)
                                    end
                                elseif transparency == 1 then
                                    WebhookSettings.SlappleLogged[modelName] = nil
                                end
                            end
                        end
                    end
                end
            end
        end
    end
    
    -- GOLDEN SLAPPLE CHECK
    if table.find(WebhookSettings.LoggedObjects, "Golden Slapple") then
        local slapplesFolder = workspace:FindFirstChild("Arena")
        if slapplesFolder then
            local island5 = slapplesFolder:FindFirstChild("island5")
            if island5 then
                local slapples = island5:FindFirstChild("Slapples")
                if slapples then
                    local goldenSlapple = slapples:FindFirstChild("GoldenSlapple")
                    if goldenSlapple and goldenSlapple:IsA("Model") then
                        local gloveMesh = goldenSlapple:FindFirstChild("Glove")
                        if gloveMesh and gloveMesh:IsA("MeshPart") then
                            local transparency = gloveMesh.Transparency
                            
                            if transparency == 0 and not WebhookSettings.GoldenSlappleLogged then
                                WebhookSettings.GoldenSlappleLogged = true
                                
                                local placeName = "Unknown"
                                pcall(function()
                                    local info = MarketplaceService:GetProductInfo(game.PlaceId)
                                    if info and info.Name then
                                        placeName = info.Name
                                    end
                                end)
                                
                                local message = string.format(
                                    "**From Webhook Log:**\n\n**Detected | Spawned:** Golden Slapple\n\n**Name:** Golden Slapple\n**Real Name:** GoldenSlapple\n**Place Name:** %s\n**Place ID:** %s\n**JobID:** %s",
                                    placeName, game.PlaceId, game.JobId
                                )
                                
                                sendWebhookMessage(message)
                            elseif transparency == 1 then
                                WebhookSettings.GoldenSlappleLogged = false
                            end
                        end
                    end
                end
            end
        end
    end
end

-- Slaps tracking function
local function checkSlaps()
    if not WebhookSettings.Enabled or WebhookSettings.SlapsThreshold == 0 or not WebhookSettings.IsSlapsTracking then return end
    
    local currentSlaps = LocalPlayer.leaderstats.Slaps.Value
    local previousSlaps = WebhookSettings.LastSlapsValue
    
    if currentSlaps >= WebhookSettings.SlapsThreshold and currentSlaps > previousSlaps then
        WebhookSettings.LastSlapsValue = currentSlaps
        
        local earnedSlaps = currentSlaps - WebhookSettings.SlapsStartValue
        
        local placeName = "Unknown"
        pcall(function()
            local info = MarketplaceService:GetProductInfo(game.PlaceId)
            if info and info.Name then
                placeName = info.Name
            end
        end)
        
        local message = string.format(
            "**Slaps Counter Log**\n\n**Name:** Slaps\n**Before Amount:** %s\n**After Amount:** %s\n**Earned Total Slaps:** +%s\n**Place Name:** %s\n**Place ID:** %s\n**JobID:** %s",
            WebhookSettings.SlapsStartValue, currentSlaps, earnedSlaps, placeName, game.PlaceId, game.JobId
        )
        
        sendWebhookMessage(message)
        
        WebhookSettings.SlapsStartValue = currentSlaps
    end
end

-- Badge tracking function
local function checkBadges()
    if not WebhookSettings.Enabled then return end
    
    -- Fetch latest badges
    local success = pcall(fetchOwnedBadges)
    if not success then return end
    
    local newBadges = {}
    
    -- Check for new badges
    for badgeId, _ in pairs(ownedBadges) do
        if not WebhookSettings.LastBadgeCheck[badgeId] then
            table.insert(newBadges, badgeId)
            WebhookSettings.LastBadgeCheck[badgeId] = true
        end
    end
    
    -- If there are new badges, send a log
    if #newBadges > 0 then
        local placeName = "Unknown"
        pcall(function()
            local info = MarketplaceService:GetProductInfo(game.PlaceId)
            if info and info.Name then
                placeName = info.Name
            end
        end)
        
        -- Find badge names
        local newBadgeNames = {}
        for _, id in pairs(newBadges) do
            for _, badge in pairs(allBadges) do
                if badge.id == id then
                    table.insert(newBadgeNames, badge.name .. " (`" .. id .. "`)")
                    break
                end
            end
        end
        
        -- Build owned list (all badges)
        local ownedList = {}
        for id, _ in pairs(WebhookSettings.LastBadgeCheck) do
            table.insert(ownedList, id)
        end
        
        -- Build missing list (all glove-rewarding badges not owned)
        local missingList = {}
        for _, badge in pairs(allBadges) do
            if badge.category == "GloveRewarding" and not ownedBadges[badge.id] then
                table.insert(missingList, badge.id)
            end
        end
        
        local message = string.format(
            "**🎉 NEW BADGE(S) UNLOCKED!**\n\n" ..
            "**Badge(s):**\n%s\n\n" ..
            "**Place Name:** %s\n" ..
            "**Place ID:** %s\n" ..
            "**JobID:** %s\n\n" ..
            "**Total Badges Unlocked:** %s\n" ..
            "**Missing Glove Badges:** %s",
            table.concat(newBadgeNames, "\n"),
            placeName,
            game.PlaceId,
            game.JobId,
            #ownedList > 0 and #ownedList .. " badges" or "None",
            #missingList > 0 and #missingList .. " badges" or "None"
        )
        
        sendWebhookMessage(message)
    end
end

-- Main loop
local function startWebhookLoop()
    while true do
        if WebhookSettings.Enabled then
            pcall(checkObjectSpawns)
            pcall(checkSlaps)
            pcall(checkNullShards)
            pcall(checkBadges)
        end
        
        local waitTime = 1
        if WebhookSettings.SendInterval == "30 Seconds" then 
            waitTime = 30
        elseif WebhookSettings.SendInterval == "1 Minute" then 
            waitTime = 60
        elseif WebhookSettings.SendInterval == "10 Minutes" then 
            waitTime = 600
        elseif WebhookSettings.SendInterval == "30 Minutes" then 
            waitTime = 1800
        elseif WebhookSettings.SendInterval == "1 Hour" then 
            waitTime = 3600
        elseif WebhookSettings.SendInterval == "5 Hours" then 
            waitTime = 18000
        elseif WebhookSettings.SendInterval == "10 Hours" then 
            waitTime = 36000
        elseif WebhookSettings.SendInterval == "1 Day" then 
            waitTime = 86400
        end
        
        task.wait(waitTime)
    end
end

-- Start the loop
task.spawn(startWebhookLoop)

-- Function to modify death sounds in a character
local function modifyDeathSounds(character)
    if not deathSoundEnabled then return end
    if not character then return end
    
    local torso = character:FindFirstChild("Torso") or character:FindFirstChild("UpperTorso")
    if not torso then return end
    
    for _, sound in pairs(torso:GetChildren()) do
        if sound:IsA("Sound") then
            local soundName = sound.Name:lower()
            if soundName:find("death") or soundName:find("die") then
                pcall(function()
                    if sound.SoundId ~= currentSoundId then
                        sound.SoundId = currentSoundId
                    end
                end)
            end
        end
    end
end

-- Function to setup listener for new sounds
local function setupSoundListener(character)
    if not character then return end
    
    local torso = character:FindFirstChild("Torso") or character:FindFirstChild("UpperTorso")
    if not torso then return end
    
    local connection = torso.ChildAdded:Connect(function(child)
        if child:IsA("Sound") and deathSoundEnabled then
            local soundName = child.Name:lower()
            if soundName:find("death") or soundName:find("die") then
                task.wait(0.1)
                pcall(function()
                    child.SoundId = currentSoundId
                end)
            end
        end
    end)
    
    table.insert(childAddedConnections, connection)
end

-- Function to start monitoring
local function startMonitoring()
    if monitorConnection then return end
    
    monitorConnection = game:GetService("RunService").Heartbeat:Connect(function()
        if deathSoundEnabled then
            local character = game.Players.LocalPlayer.Character
            if character then
                modifyDeathSounds(character)
            end
        end
    end)
    
    -- Setup listener for current character
    local character = game.Players.LocalPlayer.Character
    if character then
        setupSoundListener(character)
    end
end

-- Function to stop monitoring
local function stopMonitoring()
    cleanupConnections()
end

                        end)
                    end
                end
            end
        end
    end
end

    -- Add metatable for dynamic glove lookups
    setmetatable(gloveHits, {
        __index = function(t, key)
            -- Try to find any remote that matches the glove name
            for _, child in pairs(ReplicatedStorage:GetChildren()) do
                if child:IsA("RemoteEvent") then
                    local childName = child.Name:lower()
                    local gloveName = tostring(key):lower()
                    
                    if childName == gloveName or
                       childName == gloveName .. "hit" or
                       childName == "ht" .. gloveName or
                       childName:find(gloveName, 1, true) then
                        rawset(t, key, child) -- Cache for next time
                        return child
                    end
                end
            end
            
            -- Ultimate fallback - first available remote
            for _, child in pairs(ReplicatedStorage:GetChildren()) do
                if child:IsA("RemoteEvent") then
                    rawset(t, key, child) -- Cache for next time
                    return child
                end
            end
            
            return nil
        end
    })
    
    -- Function to get current glove
    local function getCurrentGlove()
        local success, result = pcall(function()
            if localPlayer:FindFirstChild("leaderstats") then
                local gloveValue = localPlayer.leaderstats:FindFirstChild("Glove")
                if gloveValue then
                    return gloveValue.Value
                end
            end
            return "Default"
        end)
        
        if success then
            return result
        end
        return "Default"
    end
    
    -- Function to get the remote for a glove
    local function getGloveRemote(gloveName)
        -- Try direct lookup from gloveHits
        local remote = gloveHits[gloveName]
        
        if remote and typeof(remote) == "Instance" and remote:IsA("RemoteEvent") then
            return remote
        end
        
        -- If we get here, the metatable will handle it
        return gloveHits[gloveName]
    end

            if isSlapAuraEnabled == enabled then

            end
            
            isSlapAuraEnabled = enabled
            
            if enabled then

                if slapAuraConnection then

                        return

                    local localChar = localPlayer.Character

                    for _, player in ipairs(Players:GetPlayers()) do

                            local char = player.Character
                            local humanoid = char:FindFirstChildOfClass("Humanoid")
                            local torso = char:FindFirstChild("Torso") or char:FindFirstChild("UpperTorso")

                                end
                            end
                        end

                    Name = "Slap Aura Enabled",
                    Content = "Glove: " .. currentGlove,

                    Name = "Slap Aura Disabled",
                    Content = "",

-- ServerHop (Regular Server)
Tab:AddButton({
    Name = "ServerHop (Regular Server)",
    Callback = function()
        local PlaceID = game.PlaceId
        local AllIDs = {}
        local foundAnything = ""
        local actualHour = os.date("!*t").hour
        local Deleted = false

        local success, data = pcall(function()
            return game:GetService('HttpService'):JSONDecode(readfile("NotSameServers.json"))
        end)
        if success then
            AllIDs = data
        else
            table.insert(AllIDs, actualHour)
            writefile("NotSameServers.json", game:GetService('HttpService'):JSONEncode(AllIDs))
        end

        local function TPReturner()
            local Site
            if foundAnything == "" then
                Site = game.HttpService:JSONDecode(game:HttpGet('https://games.roblox.com/v1/games/' .. PlaceID .. '/servers/Public?sortOrder=Asc&limit=100'))
            else
                Site = game.HttpService:JSONDecode(game:HttpGet('https://games.roblox.com/v1/games/' .. PlaceID .. '/servers/Public?sortOrder=Asc&limit=100&cursor=' .. foundAnything))
            end

            if Site.nextPageCursor and Site.nextPageCursor ~= "null" and Site.nextPageCursor ~= nil then
                foundAnything = Site.nextPageCursor
            end

            for i, v in pairs(Site.data) do
                local Possible = true
                local ID = tostring(v.id)
                if tonumber(v.maxPlayers) > tonumber(v.playing) then
                    for _, Existing in pairs(AllIDs) do
                        if ID == tostring(Existing) then
                            Possible = false
                            break
                        end
                    end
                    if Possible then
                        table.insert(AllIDs, ID)
                        wait()
                        pcall(function()
                            writefile("NotSameServers.json", game:GetService('HttpService'):JSONEncode(AllIDs))
                            wait()
                            game:GetService("TeleportService"):TeleportToPlaceInstance(PlaceID, ID, game.Players.LocalPlayer)
                        end)
                        wait(4)
                        break
                    end
                end
            end
        end

        local function Teleport()
            while wait() do
                pcall(function()
                    TPReturner()
                    if foundAnything ~= "" then
                        TPReturner()
                    end
                end)
            end
        end

        -- Start teleporting to new servers
        Teleport()
    end
})

local AutoEnterActive = false

    Default = "Arena",
    Options = {"Arena", "Arena Default"},

Tab:AddToggle({
    Name = "Auto Enter",
    Default = false,
    Callback = function(Value)
        AutoEnterActive = Value
        
        task.spawn(function()
            while AutoEnterActive do
                local player = game.Players.LocalPlayer
                if not player then 
                    task.wait(0.1) 
                    continue 
                end
                
                local character = player.Character
                if not character or not character.Parent then 
                    task.wait(0.5) 
                    continue 
                end
                
                local hrp = character:FindFirstChild("HumanoidRootPart")
                if not hrp then 
                    task.wait(0.1) 
                    continue 
                end
                
                -- Check if already entered
                local entered = character:FindFirstChild("entered")
                
                if not entered then
                    -- Get the correct teleport part based on mode
                    local teleportFolder = workspace:FindFirstChild("Lobby")
                    if not teleportFolder then
                        task.wait(0.5)
                        continue
                    end
                    
                    local teleportPart = nil
                    
                    if AutoEnterMode == "Arena" then
                        teleportPart = teleportFolder:FindFirstChild("Teleport1")
                    elseif AutoEnterMode == "Arena Default" then
                        teleportPart = teleportFolder:FindFirstChild("Teleport2")
                    end
                    
                    if teleportPart then
                        pcall(function()
                            -- Method 1: Teleport player directly to the teleport part
                            local touchPart = nil
                            
                            -- Find the actual part to teleport to
                            if teleportPart:IsA("BasePart") then
                                touchPart = teleportPart
                            else
                                touchPart = teleportPart:FindFirstChildWhichIsA("BasePart")
                            end
                            
                            if touchPart then
                                -- Teleport player to the teleport location
                                hrp.CFrame = touchPart.CFrame + Vector3.new(0, 3, 0)
                                task.wait(0.1)
                                
                                -- Move directly into the teleport part
                                hrp.CFrame = touchPart.CFrame
                                task.wait(0.1)
                                
                                -- Spam teleport into the part to ensure trigger
                                for i = 1, 5 do
                                    hrp.CFrame = touchPart.CFrame
                                    task.wait(0.01)
                                end
                                
                                -- Move slightly forward through the teleport
                                hrp.CFrame = touchPart.CFrame + touchPart.CFrame.LookVector * 3
                                task.wait(0.1)
                                
                                -- Teleport back and forth through the teleport
                                for i = 1, 3 do
                                    hrp.CFrame = touchPart.CFrame - touchPart.CFrame.LookVector * 2
                                    task.wait(0.05)
                                    hrp.CFrame = touchPart.CFrame + touchPart.CFrame.LookVector * 4
                                    task.wait(0.05)
                                end
                            end
                        end)
                    end
                else
                    -- Already entered, just wait
                    task.wait(1)
                end
                
                -- Wait before next attempt
                task.wait(0.5)
            end
        end)
    end
})
